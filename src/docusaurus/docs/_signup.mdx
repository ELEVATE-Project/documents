import PartialExample from './_mentored.mdx';
import Admonition from '@theme/Admonition';

1. Open <PartialExample mentored /> and click <b>Sign up</b>.

2. Enter your details:

    * <b>Name</b>
    * <b>Email</b>
    <p>
        <Admonition type="note">
        <p>Enter a valid email ID.</p>
        </Admonition>
    </p>
    * <b>Password</b>

3. Click <b>Validate OTP</b>.

4. Complete the <b>reCAPTCHA</b> verification.

5. Click <b>Generate OTP</b>. An OTP is sent to your registered email ID.

6. Enter the OTP.
<p>
    <Admonition type="tip">   
    <p>To receive a new OTP, click <b>Resend OTP</b>.</p>
    </Admonition>
</p>

7. Do the following actions:

    * To read the privacy policy, click the <b>Privacy Policy</b> link.
    * To read the terms of service, click the <b>Terms of Service</b> link.

8. Select the checkbox to agree to the terms of service and policy settings.

9. Click <b>Verify and login</b>.

10. To access the Home page and other features, create your profile.
